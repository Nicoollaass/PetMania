
<!-- QUERY POR FILIAL-->
SELECT 
	sum(QTDE),
    ID_VENDEDOR,
    USUARIO.ID_FUNCIONARIO,
    FUNCIONARIO.NOME,
    LOCACAO.DESCRICAO
    
FROM SAIDA_PRODUTO

INNER JOIN USUARIO
ON SAIDA_PRODUTO.ID_VENDEDOR = USUARIO.ID_USUARIO

INNER JOIN FUNCIONARIO
ON USUARIO.ID_FUNCIONARIO = FUNCIONARIO.ID_FUNCIONARIO

INNER JOIN LOCACAO
ON FUNCIONARIO.ID_LOCACAO = LOCACAO.ID_LOCACAO 

GROUP BY ID_VENDEDOR



<!-- QUERY POR VENDEDOR-->
SELECT 
	sum(QTDE),
    FUNCIONARIO.NOME,
FROM SAIDA_PRODUTO

INNER JOIN USUARIO
ON SAIDA_PRODUTO.ID_VENDEDOR = USUARIO.ID_USUARIO

INNER JOIN FUNCIONARIO
ON USUARIO.ID_FUNCIONARIO = FUNCIONARIO.ID_FUNCIONARIO

INNER JOIN LOCACAO
ON FUNCIONARIO.ID_LOCACAO = LOCACAO.ID_LOCACAO 

WHERE LOCACAO.ID_LOCACAO = 2
GROUP BY ID_VENDEDOR


<!-- Produto que mais vende por filial -->

SELECT 
	sum(QTDE),
    PRODUTO.DESCRICAO
    
FROM SAIDA_PRODUTO

INNER JOIN PRODUTO
ON PRODUTO.ID_PRODUTO = SAIDA_PRODUTO.ID_PRODUTO  

GROUP BY SAIDA_PRODUTO.ID_PRODUTO
